Demo.

Preview demo page

Step 1: include canim js and ease js.


    <script src="ease/ease.js"></script>
    <script src="canim.js"></script>

Step 2: define one canvas, set the width and height.


    <canvas id="myCanvas" width="720" height="1280" style="display:block;"></canvas>

Step 3: load one image, and start animation when image loaded.


    <img id="myImg" onload="main(this)" src="http://www.html5canvastutorials.com/demos/assets/darth-vader.jpg">

Step 4: define main function to add sprite to main canvas, and start the animation..


  var main = function(img){
    var canvas = document.getElementById('myCanvas');
    var c = new Canim(canvas, window.innerWidth, window.innerHeight, 10) 
    c.sprite(img).start(360, 640).scale(0,0, true).scale(1).duration(2).ease('out-elastic')
    .next(1).translate(100, 100).rotate(1).ease('out-bounce')
    .next(3).scale(0.5).move(-100, -100)
    c.start()
  }
